def count_spaces_newlines(file_name):
    space_count = 0
    newline_count = 0

    with open(file_name, 'r') as file:
        for line in file:
            space_count += line.count(' ')
            newline_count += 1

    print(f"Spaces: {space_count}")
    print(f"Newlines: {newline_count}")

count_spaces_newlines('source.txt')
